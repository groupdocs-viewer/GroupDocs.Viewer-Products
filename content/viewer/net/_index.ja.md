          ```cs
          //ビューアをインスタンス化します
          using (Viewer viewer = new Viewer("sample.docx")
          {
              //表示オプションを設定します
              HtmlViewOptions options = HtmlViewOptions.ForEmbeddedResources();
              //リソースが埋め込まれたHTMLにファイルを変換します
              viewer.View(options);
          }
          ```
